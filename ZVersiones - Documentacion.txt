// Funcion para ordenar el historial del caso de mayor a menor por fecha
ordenarHistorialCaso() {
  this.historialCaso.sort((a: any, b: any) => {
    const fechaA = new Date(a.fechaContracargo).getTime();
    const fechaB = new Date(b.fechaContracargo).getTime();

    return fechaB - fechaA; // Ordenar de mayor a menor (de más reciente a más antigua)
  });

}
